var scope="global";
function log(){
    var args=Array.prototype.join.call(arguments,",");
    //console.log(this);
    console.log(this.scope+":"+args);
}
var dog={
    scope:"dog",
    yelp:function(){
        var scope="dog.yelp";
        log('wow');

    }
};
dog.yelp(); 
dog.yelp.call(dog); 
log.apply(dog,['created'])

global:wow
global:wow (this只跟执行环境的上下文有关，这道题，log原本的作用于是全局)
dog:cretated (call的作用域问题，log的作用域被修改为dog。)
